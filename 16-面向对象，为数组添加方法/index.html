<script type="text/javascript">
	//求和
	Array.prototype.sum = function() {
		var sum = 0;
		for (var i = 0; i < this.length; i++) {
			sum +=parseFloat(this[i]);
		}
		return sum;
	}
	//求最大值
	Array.prototype.max = function() {
		var maxValue = Number.MIN_VALUE;
		for (var i = 0; i < this.length; i++) {
			if (parseFloat(this[i]) > maxValue) {
				maxValue = parseFloat(this[i]);
			}
		}
		return maxValue;
	}
	//应用
	var arr = [1,21,3,4,22,45,6,7,32];
	console.log(arr.sum()); 和：141
	console.log(arr.max()); 最大数：45
	console.log(arr.sortAscendant()); 排序：[1, 3, 4, 6, 7, 21, 22, 32, 45]
	</script>
